		//!非运算(对布尔值取反操作)
		//!!二次取反变原值
		var a = 123;
		a=!!a;//将数字类型转换为布尔
		console.log(a);
		
		//&&,两边为true则返回true。JS中为短路与左false直接返回
		//||,两边为false则返回false。JS中为短路或有true直接返回
		
		// 1，非布尔值(&&找false)
		//总结，如果第一个值是true必然返回右边的值，第一个值为false则返回第一个值
		var sum=2 && 1;
		console.log(sum);//1
		sum= 0 && 2;
		console.log(sum);//0
		sum= NaN && 0;
		// 2，非布尔值(||找true)
		//总结，第一个值是true返回第一个值，第一个值是false返回第二个值
		</script>

1，内置对象

ES标准定义的对象比如Math、Number，function等

2，宿主对象

JS运行环境、浏览器提供的对象比如BOM，DOM

3，自定义对象

开发人员自己创建的对象

2，对象的基本操作

<!DOCTYPE html>
<html>
	<head>
		<meta charset="utf-8">
		<title></title>
		<script type="text/javascript">
			//创建空的对象
			var obj=new object();
			//增加属性
			obj.name="范冰冰";
			obj.gender="男";
			obj.age=23;
			//读取属性，对象.属性名
			console.log(obj.name);
			//更改属性
			obj.age=32;
			//删除属性
			delete obj.name;
			console.log(obj.name);
            
            
            //属性名，什么乱七八糟的都可。特殊的属性名需要用：对象["属性名"]=属性值
            obj["212"]=21;
  	        console.log(obj["212"]);
            //为了使用灵活可以传变量
            obj["212"]=21;
            var n="212";
  	        console.log(obj[n]);
            //属性值，可以任意数据类型、对象
            var obj1=new object();
            obj1.name="猪八戒"
            obj.test=obj1;
            console.log(obj.test.name);
            
            //检查对象中是否含有属性返回布尔值
            console.log("name" in obj1);
		</script>
	</head>	
	<body>
	</body>
</html>
